import os
from datetime import datetime
import streamlit as st

# НАСТРОЙКА ПАРОЛЯ ДЛЯ АДМИНИСТРАТОРА
ADMIN_PASSWORD = "my_secret_password"  # ⚠️ Замените на свой пароль!

# Настройка страницы
st.set_page_config(
    page_title="Вводный инструктаж по охране труда",
    page_icon="🛡️",
    layout="centered",
)

st.title("🛡️ Охрана труда: Вводный инструктаж и Тестирование")

# Разделяем интерфейс на две вкладки
tab1, tab2 = st.tabs(["📖 1. Изучение материала", "📝 2. Прохождение теста"])

# --- ВКЛАДКА 1: ТЕКСТ ИНСТРУКТАЖА ---
with tab1:
    st.header("Программа вводного инструктажа")
    st.write(
        "Пожалуйста, внимательно ознакомьтесь с лекционным материалом перед переходом к тесту."
    )

    with st.expander("Раздел 1. Общие сведения и внутренний распорядок", expanded=True):
        st.markdown(
            """
        * **Организация:** Инструктаж обязателен для всех принимаемых на работу сотрудников до начала выполнения обязанностей.
        * **Распорядок:** Соблюдайте установленный режим рабочего времени и отдыха. Курение разрешено только в специально отведенных местах.
        """
        )

    with st.expander("Раздел 2. Опасные факторы и СИЗ"):
        st.markdown(
            """
        * **Риски:** Основными рисками в работе являются электрический ток, компьютерное излучение и невнимательность при передвижении.
        * **СИЗ:** Работник обязан правильно применять выданные ему средства индивидуальной и коллективной защиты.
        """
        )

    with st.expander("Раздел 3. Пожарная безопасность и ЧС"):
        st.markdown(
            """
        * **При пожаре:** При обнаружении задымления немедленно вызовите охрану/пожарных (101 или 112) и начните эвакуацию по знакам направления движения.
        * **Огнетушители:** В коридорах расположены порошковые/углекислотные огнетушители. Применять согласно инструкции на корпусе.
        """
        )

    with st.expander("Раздел 4. Первая помощь"):
        st.markdown(
            """
        * **Травмы:** О любой полученной микротравме или несчастном случае необходимо немедленно сообщить руководителю.
        * **Ток:** При поражении электрическим током первым делом обесточьте источник (выключите рубильник) или отбросьте провод сухим диэлектриком.
        """
        )

    st.info(
        "💡 Изучили материал? Переходите на следующую вкладку сверху, чтобы сдать тест!"
    )


# --- ВКЛАДКА 2: ТЕСТИРОВАНИЕ ---
with tab2:
    st.header("Тест для проверки знаний")

    # Форма для ввода данных сотрудника
    fio = st.text_input("Введите ваши ФИО:")
    department = st.text_input("Укажите ваш отдел / должность:")

    st.divider()

    # Вопросы теста
    q1 = st.radio(
        "Вопрос 1. В какой срок должен быть проведен вводный инструктаж?",
        [
            "А) В течение первой рабочей недели.",
            "Б) В день фактического допущения к работе, до начала выполнения обязанностей.",
            "В) В течение первого месяца работы.",
        ],
        index=None,
    )

    q2 = st.radio(
        "Вопрос 2. Что обязан сделать работник при обнаружении неисправности оборудования?",
        [
            "А) Попытаться отремонтировать самостоятельно.",
            "Б) Продолжить работу, соблюдая предельную осторожность.",
            "В) Немедленно прекратить работу, отключить оборудование и сообщить руководителю.",
        ],
        index=None,
    )

    q3 = st.radio(
        "Вопрос 3. Какое действие является первым при обнаружении пожара или задымления?",
        [
            "А) Начать спасать личные вещи и документы.",
            "Б) Вызвать пожарную охрану (101/112), сообщить руководителю и начать эвакуацию.",
            "В) Попытаться самостоятельно потушить очаг возгорания.",
        ],
        index=None,
    )

    q4 = st.radio(
        "Вопрос 4. Что необходимо сделать в первую очередь при оказании помощи пострадавшему от эл. тока?",
        [
            "А) Начать делать непрямой массаж сердца.",
            "Б) Обезопасить себя и освободить пострадавшего от действия тока (обесточить источник).",
            "В) Позвать на помощь коллег, не прикасаясь к пострадавшему.",
        ],
        index=None,
    )

    q5 = st.radio(
        "Вопрос 5. В течение какого времени нужно сообщить руководителю о микротравме?",
        [
            "А) В конце рабочей смены.",
            "Б) Только если рана начнет воспаляться.",
            "В) Немедленно в день её получения.",
        ],
        index=None,
    )

    # Кнопка отправки результатов
    if st.button("Завершить тест и отправить результат", type="primary"):
        if not fio.strip() or not department.strip():
            st.error("⚠️ Пожалуйста, заполните поля ФИО и Должность!")
        elif not all([q1, q2, q3, q4, q5]):
            st.error("⚠️ Пожалуйста, ответьте на все вопросы теста!")
        else:
            score = 0
            if q1.startswith("Б"):
                score += 1
            if q2.startswith("В"):
                score += 1
            if q3.startswith("Б"):
                score += 1
            if q4.startswith("Б"):
                score += 1
            if q5.startswith("В"):
                score += 1

            st.markdown(f"### Результат сотрудника: **{fio}**")
            st.write(f"Правильных ответов: **{score} из 5**")

            if score >= 4:
                result_status = "ЗАЧЕТ"
                st.success(
                    "🎉 ЗАЧЕТ! Инструктаж пройден успешно. Ваши результаты сохранены."
                )
            else:
                result_status = "НЕЗАЧЕТ"
                st.error(
                    "❌ НЕЗАЧЕТ. Вы допустили слишком много ошибок. Повторите материал и попробуйте снова."
                )

            # Запись результатов в файл (CSV)
            filename = "test_results.csv"
            file_exists = os.path.isfile(filename)
            current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            log_line = f"{current_time};{fio.strip()};{department.strip()};{score};{result_status}\n"

            with open(filename, "a", encoding="utf-8-sig") as file:
                if not file_exists:
                    file.write(
                        "Дата и Время;ФИО сотрудника;Отдел / Должность;Правильные ответы;Результат\n"
                    )
                file.write(log_line)


# --- 🔐 ПАНЕЛЬ АДМИНИСТРАТОРА (В САМОМ НИЗУ СТРАНИЦЫ) ---
st.write("\n" * 10)  # Делаем большой отступ вниз
st.divider()
with st.expander("🔐 Вход для инженера по ОТ / Администратора"):
    input_password = st.text_input(
        "Введите пароль для просмотра отчетов:", type="password"
    )

    if input_password == ADMIN_PASSWORD:
        st.success("Доступ разрешен!")
        filename = "test_results.csv"

        if os.path.isfile(filename):
            with open(filename, "r", encoding="utf-8-sig") as file:
                file_content = file.read()

            # Кнопка скачивания Excel-совместимого файла
            st.download_button(
                label="📥 Скачать Excel-таблицу с результатами",
                data=file_content,
                file_name="результаты_тестирования.csv",
                mime="text/csv",
            )

            # Отображаем текущие результаты текстом для быстрой проверки
            st.markdown("### Текущий список результатов:")
            st.text(file_content)
        else:
            st.info("Файл результатов пока пуст. Никто еще не проходил тест.")
    elif input_password != "":
        st.error("Неверный пароль!")
